  2. /**
  3.  * An Object Oriented Database Interface for PHP (PhpOOdb).
  4.  *
  5.  * This is a package to make coding complex, database driven applications in PHP a whole lot
  6.  * easier. Please note that this package requires PHP 5.1.0 or greater to operate. To be honest,
  7.  * you probably won't be using classes and objects much in PHP pre-5.0 anyway since so many
  8.  * features were lacking. If you really desperately need to have a PHP 4.x version of this, then
  9.  * please contact me at <b>justin dot demaris at gmail dot com</b> and I will see what I can do

  11.  *
  12.  * Vocabulary:
  13.  *
  14.  * Local vs Remote - A lot of the documentation is written from the perspective of a class that
  15.  * is extending the dbObject and using the different features. For example, if we are implementing
  16.  * the Calendar class, then the Calendar class is considered the "Local" class. When we have
  17.  * complex relationships, the Calendar class may be related to other classes, for example, a Category.
  18.  * In that case, from the perspective of the class we are implementing, the Category class is the
  19.  * "remote" class.
  20.  *
  21.  * CHANGES SINCE 0.5.1-beta:

  26.  * a wild card.

  37. /**
  38.  * Generic template for handling connections to different types of databases.
  39.  * The system currently comes with a MySql adapter by default.
  40.  *
  41.  * Extensions of this class provide the functionality to actually execute queries. Any query should
  42.  * be run by a call to runQuery. This class also provides some extra performance improvement because it
  43.  * does not make the connection to the database until a query is run. This allows us to fully utilize
  44.  * the Smarty template engine's cache feature because we can include the database connection info
  45.  * in a global file, and set up a global Database object, and then check if the page we are going
  46.  * to serve is cached. If it isn't cached, then all database queries are run as usual and the
  47.  * system works as expected. If the page IS cached, then no queries will actually be run, in which
  48.  * case, PHP will never talk to the database, saving overhead in time as well as SQL connections.
  49.  *
  50.  * So long as all of these functions are implemented, the system works completely properly with any
  51.  * sort of back end, be it MySql, PostgreSQL, flat files, or a dead badger (this may require further
  52.  * hardware drivers, see http://www.strangehorizons.com/2004/20040405/badger.shtml for help).
  53.  *
  54.  * @package DatabaseAbstract
  55.  */
  56. abstract class dbAdapter {
  57.     /**
  58.       * Holds the connection info - used during optimization to hold the connection information
  59.       * until a query is actually going to be run. Saves us from having to make a database connection
  60.       * with every page load even if we aren't going to use it.
  61.       */
  62.     protected $_connection_info = array();
  63.  
  64.     /**
  65.      * Indicator flag. Marks if we are connected or not. If we aren't, then we know
  66.      * we have to create a connection before running a query.
  67.      */
  68.     protected $_is_connected = false;
  69.     
  70.     /**
  71.      * Default Constructor.
  72.      *
  73.      * This generally does not have to be over-ridden as no connections are actually
  74.      * made right here. The actual work of the connection is done by the _connect function.
  75.      * This just stores the provided information for future reference when we DO connect.
  76.      *
  77.      * @param array $connection_info The information used to connect to the database. Will vary by adapter.
  78.      */
  79.     public function __construct($connection_info) {
  80.         // Make sure we go the proper format of input
  81.         if ( !is_array($connection_info) )
  82.             throw new Exception("Improperly Formatted Connection Information!");
  83.             
  84.         // Store connection information
  85.         $this->_connection_info = $connection_info;
  86.     }
  87.     
  88.     /**
  89.      * Runs an actual query passed to it and returns the results as an array.
  90.      *
  91.      * @param string $query The query to pass on to the database
  92.      */
  93.     public abstract function runQuery($query);
  94.     
  95.     /**
  96.      * If the connection to the database has not been made yet, then open the connection.
  97.      */
  98.     protected abstract function _connect();
  99.     
  100.     /**
  101.      * Get a row from the given table.
  102.      *
  103.      * @param string $table Name of the table to get the row from
  104.      * @param string $key Name of the field to use when selecting which row.
  105.      * @param mixed $key_value Value that $key should be.
  106.      * @return array A single associative array with each field being a key in the array. False on no such row.
  107.      */
  108.     public abstract function getRow($table, $key, $key_value);
  109.     
  110.     /**
  111.      * Write a row into the given table (Update Row)
  112.      *
  113.      * @param string $table Name of the table to update data in
  114.      * @param string $key Name of the field to use identify which field to update.
  115.      * @param array $row Associative array of data to update with. $row[$key] used as value to $key
  116.      * @return void 
  117.      */
  118.     public abstract function putRow($table, $key, $row);
  119.     
  120.     /**
  121.      * Write a row into the given table (New Row)
  122.      *
  123.      * @param string $table Name of the table to add the row into
  124.      * @param array $row Associative array of fields => values to put into that row
  125.      * @return mixed The value of unique identifier for this new row if it exists.
  126.      */
  127.     public abstract function newRow($table, $row);
  128.     
  129.     /**
  130.      * Remove a row from the given table
  131.      *
  132.      * @param string $table Name of the table to delete the row from
  133.      * @param string $key Name of the field to use when selecting which row.
  134.      * @param mixed $key_value Value that $key should be.
  135.      * @return void 
  136.      */
  137.     public abstract function delRow($table, $key, $key_value);
  138.     
  139.     /**
  140.      * Protect Variables to Prevent Injection. Whatever method is best for properly escaping values
  141.      * should be called by this function.
  142.      *
  143.      * @param string $value Value to be escaped
  144.      * @return string Escaped version of the string.
  145.      */
  146.     public abstract function escape($value);
  147.     
  148.     /**
  149.      * Get multiple rows based on a set of paramters.
  150.      *
  151.      * For example, if we want to get all rows with name equal to 'bob' and shoe size equal
  152.      * to 12, we would pass in array("name"=>"bob", "shoe_size" => 12).
  153.      *
  154.      * @param string $table The table to get the rows from
  155.      * @param array $params The restrictions on the query (see function documentation)
  156.      * @return array Returns an array of rows. Each row represented by associative array.
  157.      */
  158.     public abstract function getRows($table, $params = array());
  159.     
  160.     /**
  161.      * Delete multiple rows based on a set of restrictions.
  162.      *
  163.      * @param string $table The name of the table to delete the rows from
  164.      * @param array $params Set of restrictions on the query. (See getRows documentation)
  165.      * @return integer Number of rows deleted.
  166.      */
  167.     public abstract function delRows($table, $params);
  168.     
  169.     /**
  170.      * Update multiple rows based on a set of restrictions.
  171.      *
  172.      * The values paramter is of the same format as the $params parameter, described in
  173.      * the getRows documentation. This method takes all of the rows specifed by the second
  174.      * argument and updates the fields specified as the keys of the associative array in
  175.      * the second argument with the values associated with them.
  176.      *
  177.      * @param string $table Name of the table to update
  178.      * @param array $params Restrictions on which rows to update.
  179.      * @param array $values Associative array of which fields to update with which values
  180.      * @return integer Number of rows altered by the change.
  181.      */
  182.     public abstract function putRows($table, $params, $values);
  183. }

  186. /**
  187.  * The Database class is a level of abstraction and buffering between the DBMS and the application.
  188.  * 
  189.  * This class also provides the interface to the objects that PHP version of the database by providing
  190.  * the methods to retrive, update, and remove these object's serialized versions from the database.
  191.  *
  192.  * @package DatabaseAbstract
  193.  */
  194. class dbDatabase {
  195.     /**
  196.      * Database Adapter Object
  197.      */
  198.     private $_db = NULL;
  199.     
  200.     /**
  201.      * A place where we store references to objects created so that when another instance of
  202.      * that object is requested, we can just return the reference. This serves two purposes.
  203.      * First, it reduces database queries, which is needed since this system is very database
  204.      * intensive. Secondly, it makes it so that all Calendar objects with calendar_id = 1, for
  205.      * example, are all references to the same object. Any changes made via one reference will
  206.      * be reflected throughout the system.
  207.      */
  208.     private $_object_cache = array();
  209.     
  210.     /**
  211.      * Create this new database object with the given connection type
  212.      *
  213.      * @param string $adapter Name of the adapter to use. Currently 'MySQL' only supported.
  214.      * @param array $connection_info Info to use when connecting to the database.
  215.      * @param string $root_dir The root directory that the OOdb.php is located at. Used for finding adapters.
  216.      */
  217.     public function __construct($adapter, $connection_info, $root_dir = '') {
  218.         // Load the database adapter
  219.         require_once $root_dir."dbAdapter/{$adapter}.php";
  220.         $adapter_class = 'dbAdapter_'.$adapter;
  221.         
  222.         // Create Database Connection
  223.         $this->_db = new $adapter_class($connection_info);
  224.     }
  225.     
  226.     /**
  227.      * Retrieves an entry from the database and creates the appropriate object to bind to it.
  228.      *
  229.      * @param string $class_name Name of the class to grab an object of from the database.
  230.      * @param integer $key_value Value of the unique identifier for which object to grab.
  231.      * @return dbObject A reference to the object that matches the given value. NULL if doesn't exist.
  232.      */
  233.     public function getObject($class_name, $key_value) {
  234.         // If it's not in the cache, load it
  235.         if ( !isset($this->_object_cache[$class_name][$key_value]) ) {
  236.             $class_vars = call_user_func(array($class_name, '_dbInfo'));
  237.             $row = $this->_db->getRow($class_vars['table'], $class_vars['key'], $key_value);
  238.             if ( $row === false ) return NULL;
  239.             $this->_object_cache[$class_name][$key_value] = new $class_name($this, $row);
  240.         }
  241.         
  242.         return $this->_object_cache[$class_name][$key_value];
  243.     }
  244.  
  245.     /**
  246.      * Create a new object of the given type with a matching db entry
  247.      *
  248.      * @param string $class_name Name of the class to create an instance of (must extend dbObject)
  249.      * @param array $values Default values to assign to the object
  250.      * @return dbObject Reference to the created object.
  251.      */
  252.     public function newObject($class_name, $values) {
  253.         // Get the unique identifying information
  254.         $class_info = call_user_func(array($class_name, '_dbInfo'));
  255.         
  256.         // If no values were given, then create a blank row
  257.         if ( !is_array($values) || count($values) == 0 )
  258.             $values = $class_info['key'];
  259.  
  260.         // Create a new blank row
  261.         $id = $this->_db->newRow($class_info['table'], $values);
  262.         
  263.         // Create a new object to match to this entry
  264.         return $this->getObject($class_name, $id);
  265.     }
  266.     
  267.     /**
  268.      * Saves information about a provided object to the database.
  269.      *
  270.      * @param dbObject $object Reference to the object we want to update the database about.
  271.      * @param string $field OPTIONAL. Name of the field if we are updating a single field only.
  272.      * @return void 
  273.      */
  274.     public function putObject($object, $field = NULL) {
  275.         // Get the unique identifying information
  276.         $table = $object->getTable();
  277.         $key   = $object->getKey();
  278.  
  279.         // Create a row of information
  280.         $data = $object->_getValues();
  281.         
  282.         // If we are only updating one field, then rearrange the data
  283.         if ( $field != NULL ) 
  284.             $data = array($key => $data[$key], $field => $data[$field]);
  285.  
  286.         // Push this to the database
  287.         $this->_db->putRow($table, $key, $data);
  288.     }
  289.     
  290.     /**
  291.      * The factory itself. Loads the object from the database or creates the object if it doesn't
  292.      * exist.
  293.      *
  294.      * @param string $obj_type Name of the class to generate the object for
  295.      * @param integer $id_value OPTIONAL. Identifying value for the instance. If blank, creates new.
  296.      */
  297.     public function load($obj_type, $id_value = -1) {
  298.         // If it's a default, then create the new entry
  299.         if ( $id_value == -1 ) {
  300.             return $this->newObject($obj_type, $id_value);
  301.         }
  302.         
  303.         // Otherwise, load it from the database
  304.         else
  305.             return $this->getObject($obj_type, $id_value);
  306.     }
  307.     
  308.     /**
  309.      * Deletes an object and its corresponding database rows, along with all of the
  310.      * different entries that reference or relate to it.
  311.      *
  312.      * @param dbObject $object Reference to the object we are deleting
  313.      */
  314.     public function delObject($object) {
  315.         // Get the unique identifying information
  316.         $class_name = get_class($object);
  317.         $table = $object->getTable();
  318.         $key   = $object->getKey();
  319.         $key_value = $object->getKeyValue();
  320.         
  321.         // Tell the object to clean up and remove it's relationships
  322.         $object->_delRelations();
  323.         
  324.         // Remove the entry from the database for the object
  325.         $this->_db->delRow($table, $key, $key_value);
  326.         
  327.         // Remove the object from the local cache
  328.         unset($this->_object_cache[$class_name][$key_value]);
  329.         
  330.         // Return a NULL reference
  331.         return NULL;
  332.     }
  333.  
  334.     /**
  335.      * Get all of the objects of a particular type from the database.
  336.      *
  337.      * @param string $class_name Name of the class we are creating objects of
  338.      * @return array Array of dbObject's
  339.      */
  340.     public function getAllObjects($class_name) {
  341.         // Get information about the class
  342.         $dbInfo = call_user_func(array($class_name, _dbInfo));
  343.         $table = $dbInfo['table'];
  344.         $key   = $dbInfo['key'];
  345.  
  346.         // Get all rows from the given table
  347.         $rows = $this->_db->getRows($table);
  348.  
  349.         $objects = array();
  350.  
  351.         // Convert each row to an object
  352.         foreach ( $rows as $row ) {
  353.             // Create the object
  354.             $obj = $this->getObject($class_name, $row[$key]);
  355.             
  356.             // Add it to the array
  357.             $objects[] = $obj;
  358.         }
  359.  
  360.         // Return the list of objects
  361.         return $objects;
  362.     }
  363.  
  364.     /**
  365.      * Create a non-indexing row in a relationship table. Used for MANY-TO-MANY relationships.
  366.      *
  367.      * @param string $table_name

  371.      */
  372.     public function addListRow($table_name, $local_obj, $local_key, $remote_obj, $remote_key, $properties = array()) {

  489.  
  490.  
  491. /**
  492.  * A wrapper to contain properties and information about ONE-TO-MANY or MANY-TO-MANY relationships
  493.  * between objects.
  494.  *
  495.  * There are two ways that this class is used. First, if the list is being used to represent a
  496.  * MANY-TO-MANY relationship, then an associative table must be used in the database. In that case,
  497.  * any changes to the relationships will update this associative table. For example, if we have a
  498.  * database schema like this:
  499.  * <pre>
  500.  *        Calendars               CategoryAssign             Categories
  501.  *       ---------------         -----------------          -----------------
  502.  *       | calendar_id |  ---->  | calendar_id   |          | category_id   |
  503.  *       | cal_name    |         | category_id   |   <----- | category_name |
  504.  *       ---------------         | is_shown      |          -----------------
  505.  *                               -----------------
  506.  * </pre>
  507.  * and we want to make it so that Calendar A is now a member of Category B, then we would create
  508.  * an entry inside of the CategoryAssign table with the calendar_id of Calendar A and the category_id
  509.  * of Category B. With the dbList, we can do this using methods. For example, if we have a Calendar
  510.  * object named $CalA, and a Category object named $CatB, then we could call:
  511.  * <code>$CalA->categories->add($CatB);</code>
  512.  *
  513.  * This would update the associations in the local objects themselves as well as in the database. Note
  514.  * that when creating these associations, the properties of the relationship are set to their default
  515.  * values. In this case, is_shown would be a tiny_int (either 1 or 0) in the database and would get
  516.  * set to whatever the default is, probably 0. If we want to set is_shown to 1 when we create the
  517.  * association, then we would call:
  518.  * <code>$CalA->categories->add($CatB, array("is_shown"=>1));</code>
  519.  * We can set the values of any of the properties of the association this way. You would use the same
  520.  * method for changing the properties of an association with the update method.
  521.  *
  522.  * It is important to note that because of these properties, you cannot access one of the category objects
  523.  * directly. When you call:
  524.  * <code>$CalA->categories->getAll()</code>
  525.  * you get an array of all of the associated entries. Each entry in the array is an associative array
  526.  * with the key being the name of the property it contains. The special case is that the 'object' key
  527.  * is reserved to contain the reference to the object that is being related to. For this reason, there
  528.  * can be no properties of the relationship named 'object'.
  529.  *
  530.  * The other way this list is used is with a ONE-TO-MANY association, which occurs when we have a
  531.  * database schema like this:
  532.  * <pre>
  533.  *       Events                  Occurrences
  534.  *      ----------------        --------------
  535.  *      | event_id     |<-------| o_event_id |
  536.  *      | event_detail |        | a_date     |
  537.  *      ----------------        --------------
  538.  * </pre>
  539.  * where every occurrence belongs to an event and an event can have multiple occurrences, the Event object
  540.  * will have a ONE-TO-MANY relationship with the Occurrence. This case is significantly simpler because the
  541.  * link does not depend on a separate table and the relationship itself does not have any properties. You
  542.  * would still treat this the same way as you would if it was MANY-TO-MANY from the usability perspective.
  543.  * For example, to get a list of all of the Occurences assigned to an Event named $EventA, you would call:
  544.  * <pre>$EventA->occurrences->getAll()</pre>
  545.  *
  546.  * In order to maintain concurrency, any changes made through the dbList are reflected in the objects. For
  547.  * example, if you call:
  548.  * <code>$EventA->occurences->add($OccurenceB);</code>
  549.  * this will assign the occurrence to EventA. The Occurence object will be updated so that o_event_id is
  550.  * assigned to the event_id of EventA and the database is updated to reflect this as well.
  551.  */
  552. class dbList {
  553.     protected $_db = NULL;
  554.     protected $_assoc_table = '';
  555.     protected $_local_key = '';
  556.     protected $_remote_key = '';
  557.     
  558.     protected $_is_set = false;
  559.  
  560.     protected $_owner = NULL;
  561.     protected $_remote_class = '';
  562.     protected $_values = array();
  563.     
  564.     /**
  565.      * Constructor for a dbList. Never called directly by a user.
  566.      *
  567.      * @param dbDatabase $db Reference to the database object that all queries will run through
  568.      * @param string $assoc_table Name of the table that is used to hold relationship. Many-to-many only. Empty string otherwise.
  569.      * @param string $local_key The name of the key in the owner used to make the link
  570.      * @param dbObject $owner Reference to the object that contains this relationship.
  571.      * @param string $remote_key Name of the field in the remote class used to make the link
  572.      * @param string $remote_class Name of the class (type of object) that we are creating a link to. Must extend dbObject.
  573.      */
  574.     public function __construct ( $db, $assoc_table, $local_key, $owner, $remote_key, $remote_class ) {
  575.         // Store relationship information
  576.         $this->_db = $db;
  577.         $this->_owner = $owner;
  578.         $this->_assoc_table = $assoc_table;
  579.         $this->_local_key = $local_key;
  580.         $this->_remote_key = $remote_key;
  581.         $this->_remote_class = $remote_class;
  582.     }
  583.  
  584.     /**
  585.      * Retrieve a related object from this relationship by it's unique identifier.
  586.      *
  587.      * @param integer $id The unique identifier of the related object we want
  588.      * @return dbObject The object specified, or NULL if it doesn't exist.
  589.      */
  590.     public function get($id) {
  591.         // Load from database if not yet loaded
  592.         if ( !$this->_is_set )
  593.             $this->_getObjects();
  594.  
  595.         // If it's a one-to-many, make sure the object still points here
  596.         else if ( $this->_assoc_table == '' ) {
  597.             $remote_key = $this->_remote_key;
  598.             $local_key  = $this->_local_key;
  599.             
  600.             // If it no longer points here, then re-generate the cache
  601.             if ( $this->_values[$id]->$remote_key != $this->_owner->$local_key )
  602.                 $this->_getObjects();
  603.         }
  604.  
  605.         // Return reference to the object
  606.         if ( isset($this->_values[$id]) )
  607.             return $this->_values[$id];
  608.         else
  609.             return NULL;
  610.     }
  611.     
  612.     /**
  613.      * Populates the list of related objects in the local cache.
  614.      */
  615.     private function _getObjects() {
  616.         // Reset local array
  617.         $this->_values = array();
  618.         
  619.         // Where we get the list from depends on whether this is many-to-many or one-to-many
  620.         if ( $this->_assoc_table != '' ) {    // MANY-TO-MANY
  621.             // If we are pulling from an associative table, then pull now
  622.             $rows = $this->_db->getListRows($this->_assoc_table, $this->_owner, $this->_local_key);
  623.             $use_key = $this->_remote_key;
  624.         } else {                            // ONE-TO-MANY
  625.             // Otherwise we don't have to worry about properties of the relationship
  626.             $dbInfo = call_user_func(array($this->_remote_class, _dbInfo));
  627.             $remote_table = $dbInfo['table'];
  628.             $rows = $this->_db->getListRows($remote_table, $this->_owner, $this->_local_key);
  629.             $use_key = $dbInfo['key'];
  630.         }
  631.  
  632.         // Convert the list of entries into objects
  633.         foreach ( $rows as $row ) {
  634.             // Get the unique id used to create object
  635.             $key_value = $row[$use_key];
  636.  
  637.             // Store the properties of the relationship
  638.             $this->_values[$key_value] = $row;
  639.             unset ( $this->_values[$key_value][$this->_remote_key] );
  640.             
  641.             // Create an object entry
  642.             $this->_values[$key_value]['object'] = $this->_db->getObject($this->_remote_class, $key_value);
  643.         }
  644.  
  645.         // Update flag
  646.         $this->_is_set = true;
  647.     }
  648.     
  649.     /**
  650.      * Retrieve the whole list of related objects and properties from the local storage.
  651.      *
  652.      * @return array An array of all objects and their properties.
  653.      */
  654.     public function getAll() {
  655.         // If we haven't loaded these from the database yet, do it now
  656.         if ( !$this->_is_set )
  657.             $this->_getObjects();
  658.     
  659.         // Return the values
  660.         return $this->_values;
  661.     }
  662.  
  663.     /**
  664.      * Add a new entry
  665.      *
  666.      * @param dbObject $remote_obj 
  667.      * @param array $properties Information about the relationship
  668.      * @return boolean False if it already exists, true if succesfully added.
  669.      */
  670.     public function add($remote_obj, $properties = array()) {
  671.         // Validate the input
  672.         if ( isset($properties['object']) )
  673.             throw new Exception("Cannot have relationship property named object!");
  674.         $key = $this->_remote_key;
  675.         if ( is_null($remote_obj->$key) )
  676.             throw new Exception("Error - remote object doesn't have key $key");
  677.     
  678.         // Make sure that we have gotten the values
  679.         if ( !$this->_is_set )
  680.             $this->_getObjects();
  681.         
  682.         // Check if entry is already in the local cache
  683.         if ( isset($this->_values[$remote_obj->$key]) )
  684.             return false;
  685.         
  686.         // Add the entry to the local cache
  687.         $id = $remote_obj->$key;
  688.         $this->_values[$id]['object'] = $remote_obj;
  689.         
  690.         // Add the properties as well
  691.         foreach ( $properties as $key => $value )
  692.             $this->_values[$id][$key] = $value;
  693.         
  694.         // Update the object for one-to-many
  695.         if ( $this->_assoc_table == '' ) {
  696.             $local_key = $this->_local_key;
  697.             $remote_key = $this->_remote_key;
  698.             $local_key_val = $this->_owner->$local_key;
  699.             $remote_obj->$remote_key = $local_key_val;
  700.         }
  701.         
  702.         // Update the database for many-to-many
  703.         else {
  704.             $this->_db->addListRow($this->_assoc_table,
  705.  
  706.                                    $this->_owner,
  707.                                    $this->_local_key,
  708.                                    $remote_obj,
  709.                                    $this->_remote_key,
  710.                                    $properties);
  711.         }
  712.         
  713.         return true;
  714.     }
  715.     
  716.     /**
  717.      * Remove the given object from the local list of objects
  718.      *
  719.      * @param dbObject $remote_obj 
  720.      * @return boolean false if no matching entry. true on success.
  721.      */
  722.     public function remove($remote_obj) {
  723.         // Validate input
  724.         if ( !$remote_obj instanceof dbObject )
  725.             return false;
  726.         
  727.         // Make sure that we have gotten the values
  728.         if ( !$this->_is_set )
  729.             $this->_getObjects();
  730.         
  731.         // Check if we actually have this object in the list
  732.         $remote_key = $this->_remote_key;
  733.         if ( !isset($this->_values[$remote_obj->$remote_key]) )
  734.             return false;
  735.         
  736.         // Update the object itself for one-to-many relationships
  737.         if ( $this->_assoc_table == '' ) {
  738.             $remote_key = $this->_remote_key;
  739.             $local_key  = $this->_local_key;
  740.             
  741.             // Remove the pointer to here
  742.             $remote_obj->$remote_key = 0;
  743.         }
  744.         
  745.         // Or remove the linking row for many-to-many relationships
  746.         else {
  747.             // And update the database
  748.             $this->_db->delListRow($this->_assoc_table,
  749.                                    $this->_owner,
  750.                                    $this->_local_key,
  751.                                    $remote_obj,
  752.                                    $this->_remote_key);
  753.         }
  754.     
  755.         // And remove the local cache entry
  756.         unset($this->_values[$remote_obj->$remote_key]);
  757.     }
  758.     
  759.     /**
  760.      * Set the properties of the relationship to the given object to be the
  761.      * new given values.
  762.      *
  763.      * @param dbObject $remote_obj The object whose relationship we want to change the properties of.
  764.      * @param array $properties Associative array of the fields and values we want to change.
  765.      * @return boolean False if object not in relationship. True on successful add.
  766.      */
  767.     public function update($remote_obj, $properties) {
  768.         // Validate input
  769.         if ( !$remote_obj instanceof dbObject )
  770.             throw new Exception("Given remote object not an instance of dbObject!");
  771.         if ( isset($properties['object']) )
  772.             throw new Exception("Cannot have relationship property named object!");
  773.     
  774.         // Make sure that we have gotten the values
  775.         if ( !$this->_is_set )
  776.             $this->_getObjects();
  777.         // If it's a one-to-many, make sure the object still points here
  778.         else if ( $this->_assoc_table == '' ) {
  779.             $remote_key = $this->_remote_key;
  780.             $local_key  = $this->_local_key;
  781.             
  782.             // If it no longer points here, then re-generate the cache
  783.             if ( $this->_values[$id]->$remote_key != $this->_owner->$local_key )
  784.                 $this->_getObjects();
  785.         }
  786.  
  787.         // Make sure we have this object
  788.         $id = $remote_obj->getKeyValue();
  789.         if ( !isset($this->_values[$id]['object']) )
  790.             return false;
  791.  
  792.         // Update the object
  793.         foreach ( $properties as $key => $value )
  794.             $this->_values[$id][$key] = $value;
  795.         
  796.         // Update the database
  797.         $this->_db->putListRow($this->_assoc_table,
  798.                                $this->_owner,
  799.                                $this->_local_key,
  800.                                $remote_obj,
  801.                                $this->_remote_key,
  802.                                $properties);
  803.         return true;
  804.     }
  805. }

  807. /**
  808.  * The dbObject class is a class that allows us to bind PHP objects to a database.
  809.  * 
  810.  * This class is an abstract class that must be extended to be useful. It provides
  811.  * all of the expanded functionality required for changes to the object itself to
  812.  * be reflected immediately in the database.
  813.  *
  814.  * This class design supports pretty much any type of relational database schema, or
  815.  * at least any schema that I have ever used. If you find one that it does not support,
  816.  * then please let me know and I will do my humble best to fix that. The documentations
  817.  * are written assuming that the unique identifier is a number, but this should honestly
  818.  * support any type of value that can be represented as a string or used as the key
  819.  * of an associative array.
  820.  *
  821.  * VERY IMPORTANT NOTE - Because of the way PHP handles the concept of "static", you need
  822.  * to implement a static function in each class that extends dbObject  named _dbInfo().
  823.  * An example object that extends dbObject would look like this:
  824.  * <code>
  825.  *    class Calendar extends dbObject {
  826.  *        public static function _dbInfo() { return array("table" => 'calendar', "key" => 'calendar_id'); }
  827.  *    
  828.  *        public function __construct($db, $values) {
  829.  *            $this->_types = array("calendar_id" => "int", "title" => "text", "category_id" => "int");
  830.  *            $this->_singles[] = array('category', 'category_id', 'Calendar', 'calendar_id', true);
  831.  *        }
  832.  *    }
  833.  * </code>
  834.  * The _dbInfo function only needs to be altered by changing the values of the table and key fields
  835.  * in the array.
  836.  *
  837.  * @abstract
  838.  * @package DatabaseAbstract
  839.  */
  840. abstract class dbObject {
  841.     protected $_db = NULL;
  842.     protected static $_key = '';
  843.     protected static $_table = '';
  844.     protected $_mass_update = false;
  845.     
  846.     // Set by the user to describe the object
  847.     protected $_types   = array();
  848.     protected $_singles = array();
  849.     protected $_multis  = array();
  850.  
  851.     // properties of the object
  852.     private $_values = array();
  853.     private $_relations = array();
  854.  
  855.     /**
  856.      * Constructor to set up database connection info and type enforcement.
  857.      *
  858.      * Handles all of the set up stuff itself. You should never need to touch this except perhaps to call
  859.      * it from a lower constructor. The stuff you need to change will be in the configure() method.
  860.      *
  861.      * @param dbDatabase $db Reference to the database object that handles all database communications.
  862.      * @param string $table Name of the table that this object is bound to.
  863.      * @param string $key_name The field that is used as the unique identifier for this object.
  864.      * @param array $types An associative array of the valid fields and their data types.
  865.      * @todo Make this enforce data types
  866.      */
  867.     public function __construct($db, $values = array()) {
  868.         // set up configuration information
  869.         $this->configure();
  870.     
  871.         // Save info
  872.         $this->_db = $db;
  873.  
  874.         // Set default values
  875.         foreach ( $this->_types as $key => $type )
  876.             $this->_values[$key] = '';
  877.         
  878.         // Load in the given values
  879.         foreach ( $values as $key => &$value )
  880.             if ( isset($this->_values[$key]) )
  881.                 $this->_values[$key] = $value;
  882.         
  883.         // set up all relationships
  884.         foreach ( $this->_multis as $multi )
  885.             $this->_addMultiRelation($multi[0], $multi[1], $multi[2], $multi[3], $multi[4]);
  886.         foreach ( $this->_singles as $single )
  887.             $this->_addSingleRelation($single[0], $single[1], $single[2], $single[3], $single[4]);
  888.     }
  889.     
  890.     /**
  891.      * Used to describe the object that extends dbObject. Sets up the fields and types of data as well
  892.      * as relationships. For example, a campus that has a list of events associated with it would be
  893.      * defined like this:
  894.      * <code>
  895.      * public function configure() {
  896.      *     $this->_types = array("campus_id" => "int",
  897.      *                                  "title" => "text");
  898.      *     $this->_multis[] = array('events', 'campus_id', 'Event', 'event_id', 'wdlc_CampusAssign');
  899.      * }
  900.      * </code>
  901.      */
  902.     protected abstract function configure();
  903.     
  904.     /**
  905.      * Generates a string representation of this object.
  906.      *
  907.      * Mostly useful for debugging, this is basically giving an array printout of the row from
  908.      * the database that represents this object. Does not include related objects.
  909.      *
  910.      * @return string A string representation of this object.
  911.      */
  912.     public function __toString() {
  913.         ob_start();
  914.         echo "<pre>";
  915.         print_r($this->_values);
  916.         echo '</pre>';
  917.         $data = ob_get_contents();
  918.         ob_end_clean();
  919.         return $data;
  920.     }
  921.     
  922.     /**
  923.      * Call this function to add a "single" relationship, such as when an object is bound to one other object.
  924.      *
  925.      * For example, look at a database schema of this form:
  926.      * <pre>
  927.      *       Events                  Occurrences
  928.      *      ----------------        --------------
  929.      *      | event_id     |<-------| o_event_id |
  930.      *      | event_detail |        | a_date     |
  931.      *      ----------------        --------------
  932.      * </pre>
  933.      * where every occurrence belongs to a single event. Note that this is the same example as for the ONE-TO-MANY
  934.      * example in the dbList documentation. In this instance we need to look at the relationship from the
  935.      * perspective of the Occurence. As far as the Occurrence is concerned, there is only one Event that it has to
  936.      * relate to. It is unaware that the event may be cheating on it with other occurrences. From the Occurrence
  937.      * class we would have to call the following code to set up the relation:
  938.      * <code>
  939.      * $this->_singles[] = array('event', 'o_event_id', 'Event', 'event_id', true);
  940.      * </code>
  941.      * The constructor will automatically pass those parameters to this function.
  942.      *
  943.      * Then if we want to access the Event object that this occurrence belongs to, we can call it like this (assume
  944.      * that $OccA is an instance of Occurrence that belongs to some event):
  945.      * <code>
  946.      * echo $OccA->event;
  947.      * </code>
  948.      * This will auto-magically load the Event object with the event_id field equal to the o_event_id field in the
  949.      * Occurrence from the database and return it. It also keeps a reference in its local cache so that any other
  950.      * time afterward that you access it, it can just return it immediately. At the same time, if you would like to
  951.      * make the occurrence point to a different event, save $EventB, then you can call:
  952.      * <code>
  953.      * $OccA->event = $EventB;
  954.      * </code>
  955.      * This will automatically update the o_event_id field of the Occurrence object as well as the associated database
  956.      * row and the pointer in the Occurrence's local cache.
  957.      *
  958.      * @param string $name Name of the relationship. The field name that we will access the relationship by.
  959.      * @param string $local_key Name of the field in this object that we will use relate to the other object.
  960.      * @param string $remote_class Name of the class that the remote (related to) object will be an instance of.
  961.      * @param string $remote_key Name of the field in the remote object that will equal the local key field.
  962.      * @param string $local_link Whether this relationship value is set in this object.
  963.      * @return boolean True.
  964.      */
  965.     protected function _addSingleRelation($name, $local_key, $remote_class, $remote_key, $local_link) {
  966.         // Create the wrapper
  967.         $this->_relations[$name] = new dbObjProperty($this,
  968.                                                   $local_key,
  969.                                                   $remote_class,
  970.                                                   $remote_key,
  971.                                                   $local_link);
  972.         return true;
  973.     }
  974.     
  975.     /**
  976.      * Call this function to set up a "multi" relationship (either a ONE-TO-MANY or a MANY-TO-MANY).
  977.      *
  978.      * A MANY-TO-MANY (MTM) relationships when implemented in a database requires a third table to hold the
  979.      * relationship information. For example, the following database schema describes an MTM relationship between
  980.      * Calendars and Categories:
  981.      * <pre>
  982.      *        Calendars               CategoryAssign             Categories
  983.      *       ---------------         -----------------          -----------------
  984.      *       | calendar_id |  ---->  | calendar_id   |          | category_id   |
  985.      *       | cal_name    |         | category_id   |   <----- | category_name |
  986.      *       ---------------         | is_shown      |          -----------------
  987.      *                               -----------------
  988.      * </pre>
  989.      * In this schema, a Calendar can be assigned to multiple Categories and a Category can have multiple calendars
  990.      * assigned to it. If we want to see which calendars are assigned to a specific category, we have to look at the
  991.      * assignment table. In our code, we would have a Calendar object and a Category object. From the Calendar object
  992.      * we would access the list of Categories by calling something like this:
  993.      * <code>
  994.      *    $categories = $CalA->categories->getAll();
  995.      *    // print list of category names and whether they are going to be shown
  996.      *    foreach ( $categories as $category )
  997.      *        echo $category['object']->category_name . " - " . $category['is_shown'];
  998.      * </code>
  999.      * This would retreive a list of all of the categories along with the properties of the relationship. The objects
  1000.      * themselves would be inside the 'object' key of the associative array.
  1001.      *
  1002.      * In order to set up this relationship, we would make the following call in the object configure method:
  1003.      * <code>
  1004.      *    $this->_multis[] = array('categories', 'calendar_id', 'Category', 'category_id', 'CategoryAssign');
  1005.      * </code>
  1006.      *
  1007.      * The other relationship that this class can be used for is a ONE-TO-MANY. For example, in the following schema,
  1008.      * we have a ONE-TO-MANY (OTM) relationship between the Events and their Occurrences:
  1009.      * <pre>
  1010.      *       Events                  Occurrences
  1011.      *      ----------------        --------------
  1012.      *      | event_id     |<-------| o_event_id |
  1013.      *      | event_detail |        | a_date     |
  1014.      *      ----------------        --------------
  1015.      * </pre>
  1016.      * From the Event class we would be looking at a list of multiple Occurrences, and we would set up the relationship
  1017.      * like this:
  1018.      * <code>
  1019.      *    $this->_multis[] = array('occurrences', 'event_id', 'Occurrence', 'o_event_id');
  1020.      * </code>
  1021.      * Note that the major difference between the initiation for the ONE-TO-MANY relationship and the MANY-TO-MANY is
  1022.      * that there is no associative table in the ONE-TO-MANY one. Beyond that, accessing the entries is exactly the
  1023.      * same. For example, to display a list of all of the occurrence dates associated with Event object eventA:
  1024.      * <code>
  1025.      *    $occurrences = $eventA->occurrences->getAll();
  1026.      *    foreach ( $occurrences as $occurrence )
  1027.      *        echo $occurrence->a_date;
  1028.      * </code>
  1029.      * If you wanted to add another occurrence, for example $OccB, then you could call:
  1030.      * <code>
  1031.      *    $eventA->occurrences->add($OccB);
  1032.      * </code>
  1033.      *
  1034.      * It is important to note that in this example, we are talking about how to implement the Event class. From
  1035.      * the perspective of the Occurrence class, this is a simple "single" relationship and will be implemented
  1036.      * as described in the _addSingleRelation documentation.
  1037.      *
  1038.      * @param string $name Name of the relationship. The field name that we will access the relationship by.
  1039.      * @param string $local_key Name of the field in this object that we will use relate to the other object.
  1040.      * @param string $remote_class Name of the class that the remote (related to) object will be an instance of.
  1041.      * @param string $remote_key Name of the field in the remote object that will equal the local key field.
  1042.      * @param string $assoc_table Name of the table that contains the relationship information.
  1043.      */
  1044.     protected function _addMultiRelation($name, $local_key, $remote_class, $remote_key, $assoc_table='') {
  1045.         // Create the wrapper
  1046.         $this->_relations[$name] = new dbList($this->_db, $assoc_table, $local_key, $this, $remote_key, $remote_class);
  1047.     }
  1048.     
  1049.     /**
  1050.      * Provides database wrapped access to the values of the object.
  1051.      *

  1053.      * Whenever the user tries to access a property, for example to read "$cal->calendar_id", this function will
