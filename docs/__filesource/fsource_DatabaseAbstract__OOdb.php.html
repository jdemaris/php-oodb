  2. /**
  3.  * An Object Oriented Database Interface for PHP (PhpOOdb).
  4.  *
  5.  * This is a package to make coding complex, database driven applications in PHP a whole lot
  6.  * easier. Please note that this package requires PHP 5.1.0 or greater to operate. To be honest,
  7.  * you probably won't be using classes and objects much in PHP pre-5.0 anyway since so many
  8.  * features were lacking. If you really desperately need to have a PHP 4.x version of this, then
  9.  * please contact me at <b>justin dot demaris at gmail dot com</b> and I will see what I can do
  10.  * to port this down for you.
  11.  *
  12.  * Vocabulary:
  13.  *
  14.  * Local vs Remote - A lot of the documentation is written from the perspective of a class that
  15.  * is extending the dbObject and using the different features. For example, if we are implementing
  16.  * the Calendar class, then the Calendar class is considered the "Local" class. When we have
  17.  * complex relationships, the Calendar class may be related to other classes, for example, a Category.
  18.  * In that case, from the perspective of the class we are implementing, the Category class is the
  19.  * "remote" class.
  20.  *
  21.  * @author Justin DeMaris <justin.demaris@gmail.com>
  22.  * @package DatabaseAbstract
  23.  * @version 0.5.0-beta
  24.  *
  25.  * @todo Add support for many-to-many-to-many and beyond relationships
  26.  * @todo Add a flag to allow bulk changes to objects so that changing properties doesn't immediately update the db
  27.  * @todo Look into allowing single objects to be bound to multiple tables...
  28.  */

  30. /**
  31.  * Generic template for handling connections to different types of databases.
  32.  * The system currently comes with a MySql adapter by default.
  33.  *
  34.  * Extensions of this class provide the functionality to actually execute queries. Any query should
  35.  * be run by a call to runQuery. This class also provides some extra performance improvement because it
  36.  * does not make the connection to the database until a query is run. This allows us to fully utilize
  37.  * the Smarty template engine's cache feature because we can include the database connection info
  38.  * in a global file, and set up a global Database object, and then check if the page we are going
  39.  * to serve is cached. If it isn't cached, then all database queries are run as usual and the
  40.  * system works as expected. If the page IS cached, then no queries will actually be run, in which
  41.  * case, PHP will never talk to the database, saving overhead in time as well as SQL connections.
  42.  *
  43.  * So long as all of these functions are implemented, the system works completely properly with any
  44.  * sort of back end, be it MySql, PostgreSQL, flat files, or a dead badger (this may require further
  45.  * hardware drivers, see http://www.strangehorizons.com/2004/20040405/badger.shtml for help).
  46.  *
  47.  * @package DatabaseAbstract
  48.  */
  49. abstract class dbAdapter {
  50.     /**
  51.       * Holds the connection info - used during optimization to hold the connection information
  52.       * until a query is actually going to be run. Saves us from having to make a database connection
  53.       * with every page load even if we aren't going to use it.
  54.       */
  55.     protected $_connection_info = array();
  56.  
  57.     /**
  58.      * Indicator flag. Marks if we are connected or not. If we aren't, then we know
  59.      * we have to create a connection before running a query.
  60.      */
  61.     protected $_is_connected = false;
  62.     
  63.     /**
  64.      * Default Constructor.
  65.      *
  66.      * This generally does not have to be over-ridden as no connections are actually
  67.      * made right here. The actual work of the connection is done by the _connect function.
  68.      * This just stores the provided information for future reference when we DO connect.
  69.      *
  70.      * @param array $connection_info The information used to connect to the database. Will vary by adapter.
  71.      */
  72.     public function __construct($connection_info) {
  73.         // Make sure we go the proper format of input
  74.         if ( !is_array($connection_info) )
  75.             throw new Exception("Improperly Formatted Connection Information!");
  76.             
  77.         // Store connection information
  78.         $this->_connection_info = $connection_info;
  79.     }
  80.     
  81.     /**
  82.      * Runs an actual query passed to it and returns the results as an array.
  83.      *
  84.      * @param string $query The query to pass on to the database
  85.      */
  86.     public abstract function runQuery($query);
  87.     
  88.     /**
  89.      * If the connection to the database has not been made yet, then open the connection.
  90.      */
  91.     protected abstract function _connect();
  92.     
  93.     /**
  94.      * Get a row from the given table.
  95.      *
  96.      * @param string $table Name of the table to get the row from
  97.      * @param string $key Name of the field to use when selecting which row.
  98.      * @param mixed $key_value Value that $key should be.
  99.      * @return array A single associative array with each field being a key in the array. False on no such row.
  100.      */
  101.     public abstract function getRow($table, $key, $key_value);
  102.     
  103.     /**
  104.      * Write a row into the given table (Update Row)
  105.      *
  106.      * @param string $table Name of the table to update data in
  107.      * @param string $key Name of the field to use identify which field to update.
  108.      * @param array $row Associative array of data to update with. $row[$key] used as value to $key
  109.      * @return void 
  110.      */
  111.     public abstract function putRow($table, $key, $row);
  112.     
  113.     /**
  114.      * Write a row into the given table (New Row)
  115.      *
  116.      * @param string $table Name of the table to add the row into
  117.      * @param array $row Associative array of fields => values to put into that row
  118.      * @return mixed The value of unique identifier for this new row if it exists.
  119.      */
  120.     public abstract function newRow($table, $row);
  121.     
  122.     /**
  123.      * Remove a row from the given table
  124.      *
  125.      * @param string $table Name of the table to delete the row from
  126.      * @param string $key Name of the field to use when selecting which row.
  127.      * @param mixed $key_value Value that $key should be.
  128.      * @return void 
  129.      */
  130.     public abstract function delRow($table, $key, $key_value);
  131.     
  132.     /**
  133.      * Protect Variables to Prevent Injection. Whatever method is best for properly escaping values
  134.      * should be called by this function.
  135.      *
  136.      * @param string $value Value to be escaped
  137.      * @return string Escaped version of the string.
  138.      */
  139.     public abstract function escape($value);
  140.     
  141.     /**
  142.      * Get multiple rows based on a set of paramters.
  143.      *
  144.      * For example, if we want to get all rows with name equal to 'bob' and shoe size equal
  145.      * to 12, we would pass in array("name"=>"bob", "shoe_size" => 12).
  146.      *
  147.      * @param string $table The table to get the rows from
  148.      * @param array $params The restrictions on the query (see function documentation)
  149.      * @return array Returns an array of rows. Each row represented by associative array.
  150.      */
  151.     public abstract function getRows($table, $params);
  152.     
  153.     /**
  154.      * Delete multiple rows based on a set of restrictions.
  155.      *
  156.      * @param string $table The name of the table to delete the rows from
  157.      * @param array $params Set of restrictions on the query. (See getRows documentation)
  158.      * @return integer Number of rows deleted.
  159.      */
  160.     public abstract function delRows($table, $params);
  161.     
  162.     /**
  163.      * Update multiple rows based on a set of restrictions.
  164.      *
  165.      * The values paramter is of the same format as the $params parameter, described in
  166.      * the getRows documentation. This method takes all of the rows specifed by the second
  167.      * argument and updates the fields specified as the keys of the associative array in
  168.      * the second argument with the values associated with them.
  169.      *
  170.      * @param string $table Name of the table to update
  171.      * @param array $params Restrictions on which rows to update.
  172.      * @param array $values Associative array of which fields to update with which values
  173.      * @return integer Number of rows altered by the change.
  174.      */
  175.     public abstract function putRows($table, $params, $values);
  176. }

  179. /**
  180.  * The Database class is a level of abstraction and buffering between the DBMS and the application.
  181.  * 
  182.  * This class also provides the interface to the objects that PHP version of the database by providing
  183.  * the methods to retrive, update, and remove these object's serialized versions from the database.
  184.  *
  185.  * @package DatabaseAbstract
  186.  */
  187. class dbDatabase {
  188.     /**
  189.      * Database Adapter Object
  190.      */
  191.     private $_db = NULL;
  192.     
  193.     /**
  194.      * A place where we store references to objects created so that when another instance of
  195.      * that object is requested, we can just return the reference. This serves two purposes.
  196.      * First, it reduces database queries, which is needed since this system is very database
  197.      * intensive. Secondly, it makes it so that all Calendar objects with calendar_id = 1, for
  198.      * example, are all references to the same object. Any changes made via one reference will
  199.      * be reflected throughout the system.
  200.      */
  201.     private $_object_cache = array();
  202.     
  203.     /**
  204.      * Create this new database object with the given connection type
  205.      *
  206.      * @param string $adapter Name of the adapter to use. Currently 'MySQL' only supported.
  207.      * @param array $connection_info Info to use when connecting to the database.
  208.      * @param string $root_dir The root directory that the OOdb.php is located at. Used for finding adapters.
  209.      */
  210.     public function __construct($adapter, $connection_info, $root_dir = '') {
  211.         // Load the database adapter
  212.         require_once $root_dir."dbAdapter/{$adapter}.php";
  213.         $adapter_class = 'dbAdapter_'.$adapter;
  214.         
  215.         // Create Database Connection
  216.         $this->_db = new $adapter_class($connection_info);
  217.     }
  218.     
  219.     /**
  220.      * Retrieves an entry from the database and creates the appropriate object to bind to it.
  221.      *
  222.      * @param string $class_name Name of the class to grab an object of from the database.
  223.      * @param integer $key_value Value of the unique identifier for which object to grab.
  224.      * @return dbObject A reference to the object that matches the given value. NULL if doesn't exist.
  225.      */
  226.     public function getObject($class_name, $key_value) {
  227.         // If it's not in the cache, load it
  228.         if ( !isset($this->_object_cache[$class_name][$key_value]) ) {
  229.             $class_vars = call_user_func(array($class_name, '_dbInfo'));
  230.             $row = $this->_db->getRow($class_vars['table'], $class_vars['key'], $key_value);
  231.             if ( $row === false ) return NULL;
  232.             $this->_object_cache[$class_name][$key_value] = new $class_name($this, $row);
  233.         }
  234.         
  235.         return $this->_object_cache[$class_name][$key_value];
  236.     }
  237.  
  238.     /**
  239.      * Create a new object of the given type with a matching db entry
  240.      *
  241.      * @param string $class_name Name of the class to create an instance of (must extend dbObject)
  242.      * @param array $values Default values to assign to the object
  243.      * @return dbObject Reference to the created object.
  244.      */
  245.     public function newObject($class_name, $values) {
  246.         // Get the unique identifying information
  247.         $class_info = call_user_func(array($class_name, '_dbInfo'));
  248.  
  249.         // Create a new blank row
  250.         $id = $this->_db->newRow($class_info['table'], $values);
  251.         
  252.         // Create a new object to match to this entry
  253.         return $this->getObject($class_name, $id);
  254.     }
  255.     
  256.     /**
  257.      * Saves information about a provided object to the database.
  258.      *
  259.      * @param dbObject $object Reference to the object we want to update the database about.
  260.      * @return void 
  261.      */
  262.     public function putObject($object) {
  263.         // Get the unique identifying information
  264.         $table = $object->getTable();
  265.         $key   = $object->getKey();
  266.  
  267.         // Create a row of information
  268.         $data = $object->_getValues();
  269.         
  270.         // Push this to the database
  271.         $this->_db->putRow($table, $key, $data);
  272.     }
  273.     
  274.     /**
  275.      * Deletes an object and its corresponding database rows, along with all of the
  276.      * different entries that reference or relate to it.
  277.      *
  278.      * @param dbObject $object Reference to the object we are deleting
  279.      */
  280.     public function delObject($object) {
  281.         // Get the unique identifying information
  282.         $class_name = get_class($object);
  283.         $table = $object->getTable();
  284.         $key   = $object->getKey();
  285.         $key_value = $object->getKeyValue();
  286.         
  287.         // Tell the object to clean up and remove it's relationships
  288.         $object->_delRelations();
  289.         
  290.         // Remove the entry from the database for the object
  291.         $this->_db->delRow($table, $key, $key_value);
  292.         
  293.         // Remove the object from the local cache
  294.         unset($this->_object_cache[$class_name][$key_value]);
  295.         
  296.         // Return a NULL reference
  297.         return NULL;
  298.     }
  299.     
  300.     /**
  301.      * Create a non-indexing row in a relationship table. Used for MANY-TO-MANY relationships.
  302.      *
  303.      * @param string $table_name 
  304.      * @param dbObject $local_obj 
  305.      * @param dbObject $remote_obj 
  306.      * @param array $properties OPTIONAL
  307.      */
  308.     public function addListRow($table_name, $local_obj, $local_key, $remote_obj, $remote_key, $properties = array()) {
  309.         // validate input
  310.         if ( !$local_obj instanceof dbObject || !$remote_obj instanceof dbObject )
  311.             throw new Exception("Local and remote objects must be dbObjects");
  312.     
  313.         // Combine it all into one key/value array
  314.         $properties[$local_key] = $local_obj->$local_key;
  315.         $properties[$remote_key] = $remote_obj->$remote_key;
  316.     
  317.         // Create the row
  318.         $this->_db->newRow($table_name, $properties);
  319.         
  320.         // Return success
  321.         return true;
  322.     }
  323.     
  324.     /**
  325.      * Retrieve a list of remote rows to generate relationship objects with.
  326.      *
  327.      * @param string $assoc_table The table used in the association
  328.      * @param dbObject $local_obj The object that we want to use as the base for the search
  329.      * @param string $local_key The field in the local_obj to use as the key for the search
  330.      * @return array An array of rows.
  331.      */
  332.     public function getListRows($assoc_table, $local_obj, $local_key) {
  333.         $params = array($local_key => $local_obj->$local_key);
  334.         return $this->_db->getRows($assoc_table, $params);
  335.     }
  336.     
  337.     /**
  338.      * Remove the list row entry from the relationship table
  339.      *
  340.      * @param string $table_name Name of the table to remove the row from
  341.      * @param dbObject $local_obj Reference to the object used as the base reference
  342.      * @param string $local_key The field of the local_obj that is used as the key in the selection
  343.      * @param dbObject $remote_obj Reference to the object we are removing the association with
  344.      * @param string $remote_key The field of the remote_obj that is used as the key in the selection
  345.      * @return integer Number of rows deleted.
  346.      */
  347.     public function delListRow($table_name, $local_obj, $local_key, $remote_obj, $remote_key) {
  348.         // validate input
  349.         if ( !$local_obj instanceof dbObject || !$remote_obj instanceof dbObject )
  350.             throw new Exception("Local and remote objects must be dbObjects");
  351.  
  352.         // Set up local restrictions
  353.         $key = $local_key;
  354.         $key_value = $local_obj->$local_key;
  355.         $params[$key] = $key_value;
  356.         
  357.         // Set up remote restrictions
  358.         $key = $remote_key;
  359.         $key_value = $remote_obj->$remote_key;
  360.         $params[$key] = $key_value;
  361.         
  362.         // Remove the row
  363.         $num_deleted = $this->_db->delRows($table_name, $params);
  364.         
  365.         // Return success
  366.         return $num_deleted;
  367.     }
  368.     
  369.     /**
  370.      * Update the list row entry in the relationship table with new properties
  371.      *
  372.      * @param string $table_name Name of the table to update the row in
  373.      * @param dbObject $local_obj Reference to the object used as the base reference
  374.      * @param string $local_key The field of the local_obj that is used as the key in the selection
  375.      * @param dbObject $remote_obj Reference to the object we are altering the association with
  376.      * @param string $remote_key The field of the remote_obj that is used as the key in the selection
  377.      * @param array $properties Associative array containing the properties of the relationship to update
  378.      * @return integer Number of rows affected by the change.
  379.      */
  380.     public function putListRow($table_name, $local_obj, $local_key, $remote_obj, $remote_key, $properties) {
  381.         // validate input
  382.         if ( !$local_obj instanceof dbObject || !$remote_obj instanceof dbObject )
  383.             throw new Exception("Local and remote objects must be dbObjects");
  384.  
  385.         // Set up local restrictions
  386.         $key = $local_key;
  387.         $key_value = $local_obj->$local_key;
  388.         $params[$key] = $key_value;
  389.         
  390.         // Set up remote restrictions
  391.         $key = $remote_key;
  392.         $key_value = $remote_obj->$remote_key;
  393.         $params[$key] = $key_value;
  394.         
  395.         // Update the row
  396.         $num_editted = $this->_db->putRows($table_name, $params, $properties);
  397.         
  398.         // Return success
  399.         return $num_editted;
  400.     }
  401.     
  402.     /**
  403.      * Allows direct access to the database connection. Passes queries directly to the adapter.
  404.      *
  405.      * @param string $query The query to run on the database system.
  406.      * @return array An array of associative arrays representing rows as a result of the query.
  407.      */
  408.     public function runQuery($query) {
  409.         return $this->_db->runQuery($query);
  410.     }
  411. }
  412.  
  413. /**
  414.  * A wrapper to contain properties and information about ONE-TO-MANY or MANY-TO-MANY relationships
  415.  * between objects.
  416.  *
  417.  * There are two ways that this class is used. First, if the list is being used to represent a
  418.  * MANY-TO-MANY relationship, then an associative table must be used in the database. In that case,
  419.  * any changes to the relationships will update this associative table. For example, if we have a
  420.  * database schema like this:
  421.  * <pre>
  422.  *        Calendars               CategoryAssign             Categories
  423.  *       ---------------         -----------------          -----------------
  424.  *       | calendar_id |  ---->  | calendar_id   |          | category_id   |
  425.  *       | cal_name    |         | category_id   |   <----- | category_name |
  426.  *       ---------------         | is_shown      |          -----------------
  427.  *                               -----------------
  428.  * </pre>
  429.  * and we want to make it so that Calendar A is now a member of Category B, then we would create
  430.  * an entry inside of the CategoryAssign table with the calendar_id of Calendar A and the category_id
  431.  * of Category B. With the dbList, we can do this using methods. For example, if we have a Calendar
  432.  * object named $CalA, and a Category object named $CatB, then we could call:
  433.  * <code>$CalA->categories->add($CatB);</code>
  434.  *
  435.  * This would update the associations in the local objects themselves as well as in the database. Note
  436.  * that when creating these associations, the properties of the relationship are set to their default
  437.  * values. In this case, is_shown would be a tiny_int (either 1 or 0) in the database and would get
  438.  * set to whatever the default is, probably 0. If we want to set is_shown to 1 when we create the
  439.  * association, then we would call:
  440.  * <code>$CalA->categories->add($CatB, array("is_shown"=>1));</code>
  441.  * We can set the values of any of the properties of the association this way. You would use the same
  442.  * method for changing the properties of an association with the update method.
  443.  *
  444.  * It is important to note that because of these properties, you cannot access one of the category objects
  445.  * directly. When you call:
  446.  * <code>$CalA->categories->getAll()</code>
  447.  * you get an array of all of the associated entries. Each entry in the array is an associative array
  448.  * with the key being the name of the property it contains. The special case is that the 'object' key
  449.  * is reserved to contain the reference to the object that is being related to. For this reason, there
  450.  * can be no properties of the relationship named 'object'.
  451.  *
  452.  * The other way this list is used is with a ONE-TO-MANY association, which occurs when we have a
  453.  * database schema like this:
  454.  * <pre>
  455.  *       Events                  Occurrences
  456.  *      ----------------        --------------
  457.  *      | event_id     |<-------| o_event_id |
  458.  *      | event_detail |        | a_date     |
  459.  *      ----------------        --------------
  460.  * </pre>
  461.  * where every occurrence belongs to an event and an event can have multiple occurrences, the Event object
  462.  * will have a ONE-TO-MANY relationship with the Occurrence. This case is significantly simpler because the
  463.  * link does not depend on a separate table and the relationship itself does not have any properties. You
  464.  * would still treat this the same way as you would if it was MANY-TO-MANY from the usability perspective.
  465.  * For example, to get a list of all of the Occurences assigned to an Event named $EventA, you would call:
  466.  * <pre>$EventA->occurrences->getAll()</pre>
  467.  *
  468.  * In order to maintain concurrency, any changes made through the dbList are reflected in the objects. For
  469.  * example, if you call:
  470.  * <code>$EventA->occurences->add($OccurenceB);</code>
  471.  * this will assign the occurrence to EventA. The Occurence object will be updated so that o_event_id is
  472.  * assigned to the event_id of EventA and the database is updated to reflect this as well.
  473.  */
  474. class dbList {
  475.     protected $_db = NULL;
  476.     protected $_assoc_table = '';
  477.     protected $_local_key = '';
  478.     protected $_remote_key = '';
  479.     
  480.     protected $_is_set = false;
  481.  
  482.     protected $_owner = NULL;
  483.     protected $_remote_class = '';
  484.     protected $_values = array();
  485.     
  486.     /**
  487.      * Constructor for a dbList. Never called directly by a user.
  488.      *
  489.      * @param dbDatabase $db Reference to the database object that all queries will run through
  490.      * @param string $assoc_table Name of the table that is used to hold relationship. Many-to-many only. Empty string otherwise.
  491.      * @param string $local_key The name of the key in the owner used to make the link
  492.      * @param dbObject $owner Reference to the object that contains this relationship.
  493.      * @param string $remote_key Name of the field in the remote class used to make the link
  494.      * @param string $remote_class Name of the class (type of object) that we are creating a link to. Must extend dbObject.
  495.      */
  496.     public function __construct ( $db, $assoc_table, $local_key, $owner, $remote_key, $remote_class ) {
  497.         // Store relationship information
  498.         $this->_db = $db;
  499.         $this->_owner = $owner;
  500.         $this->_assoc_table = $assoc_table;
  501.         $this->_local_key = $local_key;
  502.         $this->_remote_key = $remote_key;
  503.         $this->_remote_class = $remote_class;
  504.     }
  505.  
  506.     /**
  507.      * Retrieve a related object from this relationship by it's unique identifier.
  508.      *
  509.      * @param integer $id The unique identifier of the related object we want
  510.      * @return dbObject The object specified, or NULL if it doesn't exist.
  511.      */
  512.     public function get($id) {
  513.         // Load from database if not yet loaded
  514.         if ( !$this->_is_set )
  515.             $this->_getObjects();
  516.  
  517.         // If it's a one-to-many, make sure the object still points here
  518.         else if ( $this->_assoc_table == '' ) {
  519.             $remote_key = $this->_remote_key;
  520.             $local_key  = $this->_local_key;
  521.             
  522.             // If it no longer points here, then re-generate the cache
  523.             if ( $this->_values[$id]->$remote_key != $this->_owner->$local_key )
  524.                 $this->_getObjects();
  525.         }
  526.  
  527.         // Return reference to the object
  528.         if ( isset($this->_values[$id]) )
  529.             return $this->_values[$id];
  530.         else
  531.             return NULL;
  532.     }
  533.     
  534.     /**
  535.      * Populates the list of related objects in the local cache.
  536.      */
  537.     private function _getObjects() {
  538.         // Reset local array
  539.         $this->_values = array();
  540.         
  541.         // Where we get the list from depends on whether this is many-to-many or one-to-many
  542.         if ( $this->_assoc_table != '' ) {    // MANY-TO-MANY
  543.             // If we are pulling from an associative table, then pull now
  544.             $rows = $this->_db->getListRows($this->_assoc_table, $this->_owner, $this->_local_key);
  545.             $use_key = $this->_remote_key;
  546.         } else {                            // ONE-TO-MANY
  547.             // Otherwise we don't have to worry about properties of the relationship
  548.             $dbInfo = call_user_func(array($this->_remote_class, _dbInfo));
  549.             $remote_table = $dbInfo['table'];
  550.             $rows = $this->_db->getListRows($remote_table, $this->_owner, $this->_local_key);
  551.             $use_key = $dbInfo['key'];
  552.         }
  553.  
  554.         // Convert the list of entries into objects
  555.         foreach ( $rows as $row ) {
  556.             // Get the unique id used to create object
  557.             $key_value = $row[$use_key];
  558.  
  559.             // Store the properties of the relationship
  560.             $this->_values[$key_value] = $row;
  561.             unset ( $this->_values[$key_value][$this->_remote_key] );
  562.             
  563.             // Create an object entry
  564.             $this->_values[$key_value]['object'] = $this->_db->getObject($this->_remote_class, $key_value);
  565.         }
  566.  
  567.         // Update flag
  568.         $this->_is_set = true;
  569.     }
  570.     
  571.     /**
  572.      * Retrieve the whole list of related objects and properties from the local storage.
  573.      *
  574.      * @return array An array of all objects and their properties.
  575.      */
  576.     public function getAll() {
  577.         // If we haven't loaded these from the database yet, do it now
  578.         if ( !$this->_is_set )
  579.             $this->_getObjects();
  580.     
  581.         // Return the values
  582.         return $this->_values;
  583.     }
  584.  
  585.     /**
  586.      * Add a new entry
  587.      *
  588.      * @param dbObject $remote_obj 
  589.      * @param array $properties Information about the relationship
  590.      * @return boolean False if it already exists, true if succesfully added.
  591.      */
  592.     public function add($remote_obj, $properties = array()) {
  593.         // Validate the input
  594.         if ( isset($properties['object']) )
  595.             throw new Exception("Cannot have relationship property named object!");
  596.         $key = $this->_remote_key;
  597.         if ( is_null($remote_obj->$key) )
  598.             throw new Exception("Error - remote object doesn't have key $key");
  599.     
  600.         // Make sure that we have gotten the values
  601.         if ( !$this->_is_set )
  602.             $this->_getObjects();
  603.         
  604.         // Check if entry is already in the local cache
  605.         if ( isset($this->_values[$remote_obj->$key]) )
  606.             return false;
  607.         
  608.         // Add the entry to the local cache
  609.         $id = $remote_obj->$key;
  610.         $this->_values[$id]['object'] = $remote_obj;
  611.         
  612.         // Add the properties as well
  613.         foreach ( $properties as $key => $value )
  614.             $this->_values[$id][$key] = $value;
  615.         
  616.         // Update the object for one-to-many
  617.         if ( $this->_assoc_table == '' ) {
  618.             $local_key = $this->_local_key;
  619.             $remote_key = $this->_remote_key;
  620.             $local_key_val = $this->_owner->$local_key;
  621.             $remote_obj->$remote_key = $local_key_val;
  622.         }
  623.         
  624.         // Update the database for many-to-many
  625.         else {
  626.             $this->_db->addListRow($this->_assoc_table,
  627.  
  628.                                    $this->_owner,
  629.                                    $this->_local_key,
  630.                                    $remote_obj,
  631.                                    $this->_remote_key,
  632.                                    $properties);
  633.         }
  634.         
  635.         return true;
  636.     }
  637.     
  638.     /**
  639.      * Remove the given object from the local list of objects
  640.      *
  641.      * @param dbObject $remote_obj 
  642.      * @return boolean false if no matching entry. true on success.
  643.      */
  644.     public function remove($remote_obj) {
  645.         // Validate input
  646.         if ( !$remote_obj instanceof dbObject )
  647.             throw new Exception("remote object must be a dbObject");
  648.         
  649.         // Make sure that we have gotten the values
  650.         if ( !$this->_is_set )
  651.             $this->_getObjects();
  652.         
  653.         // Check if we actually have this object in the list
  654.         $remote_key = $this->_remote_key;
  655.         if ( !isset($this->_values[$remote_obj->$remote_key]) )
  656.             return false;
  657.         
  658.         // Update the object itself for one-to-many relationships
  659.         if ( $this->_assoc_table == '' ) {
  660.             $remote_key = $this->_remote_key;
  661.             $local_key  = $this->_local_key;
  662.             
  663.             // Remove the pointer to here
  664.             $remote_obj->$remote_key = 0;
  665.         }
  666.         
  667.         // Or remove the linking row for many-to-many relationships
  668.         else {
  669.             // And update the database
  670.             $this->_db->delListRow($this->_assoc_table,
  671.                                    $this->_owner,
  672.                                    $this->_local_key,
  673.                                    $remote_obj,
  674.                                    $this->_remote_key);
  675.         }
  676.     
  677.         // And remove the local cache entry
  678.         unset($this->_values[$remote_obj->$remote_key]);
  679.     }
  680.     
  681.     /**
  682.      * Set the properties of the relationship to the given object to be the
  683.      * new given values.
  684.      *
  685.      * @param dbObject $remote_obj The object whose relationship we want to change the properties of.
  686.      * @param array $properties Associative array of the fields and values we want to change.
  687.      * @return boolean False if object not in relationship. True on successful add.
  688.      */
  689.     public function update($remote_obj, $properties) {
  690.         // Validate input
  691.         if ( !$remote_obj instanceof dbObject )
  692.             throw new Exception("Given remote object not an instance of dbObject!");
  693.         if ( isset($properties['object']) )
  694.             throw new Exception("Cannot have relationship property named object!");
  695.     
  696.         // Make sure that we have gotten the values
  697.         if ( !$this->_is_set )
  698.             $this->_getObjects();
  699.         // If it's a one-to-many, make sure the object still points here
  700.         else if ( $this->_assoc_table == '' ) {
  701.             $remote_key = $this->_remote_key;
  702.             $local_key  = $this->_local_key;
  703.             
  704.             // If it no longer points here, then re-generate the cache
  705.             if ( $this->_values[$id]->$remote_key != $this->_owner->$local_key )
  706.                 $this->_getObjects();
  707.         }
  708.  
  709.         // Make sure we have this object
  710.         $id = $remote_obj->getKeyValue();
  711.         if ( !isset($this->_values[$id]['object']) )
  712.             return false;
  713.  
  714.         // Update the object
  715.         foreach ( $properties as $key => $value )
  716.             $this->_values[$id][$key] = $value;
  717.         
  718.         // Update the database
  719.         $this->_db->putListRow($this->_assoc_table,
  720.                                $this->_owner,
  721.                                $this->_local_key,
  722.                                $remote_obj,
  723.                                $this->_remote_key,
  724.                                $properties);
  725.         return true;
  726.     }
  727. }
  728.  
  729. /**
  730.  * The dbObject class is a class that allows us to bind PHP objects to a database.
  731.  * 
  732.  * This class is an abstract class that must be extended to be useful. It provides
  733.  * all of the expanded functionality required for changes to the object itself to
  734.  * be reflected immediately in the database.
  735.  *
  736.  * Any class that extends this one must have a constructor signature of the following format:
  737.  *
  738.  *    public function __construct($db, $row)
  739.  *
  740.  * This class design supports pretty much any type of relational database schema, or
  741.  * at least any schema that I have ever used. If you find one that it does not support,
  742.  * then please let me know and I will do my humble best to fix that. The documentations
  743.  * are written assuming that the unique identifier is a number, but this should honestly
  744.  * support any type of value that can be represented as a string or used as the key
  745.  * of an associative array.
  746.  *
  747.  * VERY IMPORTANT NOTE - Because of the way PHP handles the concept of "static", you need
  748.  * to implement two static functions in each class that extends dbObject, one named _dbInfo()
  749.  * and one named load. An example object that extends dbObject would look like this:
  750.  * <code>
  751.  *    class Calendar extends dbObject {
  752.  *        public static function _dbInfo() { return array("table" => 'calendar', "key" => 'calendar_id'); }
  753.  *    
  754.  *        public function __construct($db, $values) {
  755.  *            $types = array("calendar_id" => "int", "title" => "text", "category_id" => "int");
  756.  *            parent::__construct($db, $types, $values);
  757.  *            $this->_addSingleRelation('category', 'category_id', 'Calendar', 'calendar_id', true);
  758.  *        }
  759.  *    
  760.  *        public static function load($db, $id) {
  761.  *            return $db->getObject(__CLASS__, $id);
  762.  *        }
  763.  *    }
  764.  * </code>
  765.  * The load function can be copied and pasted into every class as it is exactly the same. The _dbInfo
  766.  * function only needs to be altered by changing the values of the table and key fields in the array.
  767.  *
  768.  * @abstract
  769.  * @package DatabaseAbstract
  770.  */
  771. abstract class dbObject {
  772.     protected $_db = NULL;
  773.     protected static $_key = '';
  774.     protected static $_table = '';
  775.     protected $_mass_update = false;
  776.  
  777.     private $_values = array();
  778.     private $_types  = array();
  779.     private $_relations = array();
  780.  
  781.     /**
  782.      * Constructor to set up database connection info and type enforcement.
  783.      *
  784.      * For example, if the Calendars table is formatted like this:
  785.      * <pre>
  786.      *        Calendars
  787.      *       ---------------
  788.      *       | calendar_id |
  789.      *       | cal_name    |
  790.      *       ---------------
  791.      * </pre>
  792.      * then we would write the constructor on the Calendar object like this:
  793.      * <code>
  794.      *    $types = array('calendar_id'=>'integer', 'cal_name' => 'text');
  795.      *    parent::__construct($db, $types);
  796.      * </code>
  797.      *
  798.      * @param dbDatabase $db Reference to the database object that handles all database communications.
  799.      * @param string $table Name of the table that this object is bound to.
  800.      * @param string $key_name The field that is used as the unique identifier for this object.
  801.      * @param array $types An associative array of the valid fields and their data types.
  802.      * @todo Make this enforce data types
  803.      */
  804.     public function __construct($db, $types, $values = array()) {
  805.         // Save info
  806.         $this->_db = $db;
  807.         $this->_types = $types;
  808.  
  809.         // Set default values
  810.         foreach ( $types as $key => $type )
  811.             $this->_values[$key] = '';
  812.         
  813.         // Load in the given values
  814.         foreach ( $values as $key => &$value )
  815.             if ( isset($this->_values[$key]) )
  816.                 $this->_values[$key] = $value;
  817.     }
  818.     
  819.     /**
  820.      * Generates a string representation of this object.
  821.      *
  822.      * Mostly useful for debugging, this is basically giving an array printout of the row from
  823.      * the database that represents this object. Does not include related objects.
  824.      *
  825.      * @return string A string representation of this object.
  826.      */
  827.     public function __toString() {
  828.         ob_start();
  829.         echo "<pre>";
  830.         print_r($this->_values);
  831.         echo '</pre>';
  832.         $data = ob_get_contents();
  833.         ob_end_clean();
  834.         return $data;
  835.     }
  836.     
  837.     /**
  838.      * Call this function to add a "single" relationship, such as when an object is bound to one other object.
  839.      *
  840.      * For example, look at a database schema of this form:
  841.      * <pre>
  842.      *       Events                  Occurrences
  843.      *      ----------------        --------------
  844.      *      | event_id     |<-------| o_event_id |
  845.      *      | event_detail |        | a_date     |
  846.      *      ----------------        --------------
  847.      * </pre>
  848.      * where every occurrence belongs to a single event. Note that this is the same example as for the ONE-TO-MANY
  849.      * example in the dbList documentation. In this instance we need to look at the relationship from the
  850.      * perspective of the Occurence. As far as the Occurrence is concerned, there is only one Event that it has to
  851.      * relate to. It is unaware that the event may be cheating on it with other occurrences. From the Occurrence
  852.      * class we would have to call the following constructor code:
  853.      * <code>
  854.      * $this->_addSingleRelation('event', 'o_event_id', 'Event', 'event_id', true);
  855.      * </code>
  856.      * Then if we want to access the Event object that this occurrence belongs to, we can call it like this (assume
  857.      * that $OccA is an instance of Occurrence that belongs to some event):
  858.      * <code>
  859.      * echo $OccA->event;
  860.      * </code>
  861.      * This will auto-magically load the Event object with the event_id field equal to the o_event_id field in the
  862.      * Occurrence from the database and return it. It also keeps a reference in its local cache so that any other
  863.      * time afterward that you access it, it can just return it immediately. At the same time, if you would like to
  864.      * make the occurrence point to a different event, save $EventB, then you can call:
  865.      * <code>
  866.      * $OccA->event = $EventB;
  867.      * </code>
  868.      * This will automatically update the o_event_id field of the Occurrence object as well as the associated database
  869.      * row and the pointer in the Occurrence's local cache.
  870.      *
  871.      * @param string $name Name of the relationship. The field name that we will access the relationship by.
  872.      * @param string $local_key Name of the field in this object that we will use relate to the other object.
  873.      * @param string $remote_class Name of the class that the remote (related to) object will be an instance of.
  874.      * @param string $remote_key Name of the field in the remote object that will equal the local key field.
  875.      * @param string $local_link Whether this relationship value is set in this object.
  876.      * @return boolean True.
  877.      */
  878.     protected function _addSingleRelation($name, $local_key, $remote_class, $remote_key, $local_link) {
  879.         // Create the wrapper
  880.         $this->_relations[$name] = new dbObjProperty($this,
  881.                                                   $local_key,
  882.                                                   $remote_class,
  883.                                                   $remote_key,
  884.                                                   $local_link);
  885.         return true;
  886.     }
  887.     
  888.     /**
  889.      * Call this function to set up a "multi" relationship (either a ONE-TO-MANY or a MANY-TO-MANY).
  890.      *
  891.      * A MANY-TO-MANY (MTM) relationships when implemented in a database requires a third table to hold the
  892.      * relationship information. For example, the following database schema describes an MTM relationship between
  893.      * Calendars and Categories:
  894.      * <pre>
  895.      *        Calendars               CategoryAssign             Categories
  896.      *       ---------------         -----------------          -----------------
  897.      *       | calendar_id |  ---->  | calendar_id   |          | category_id   |
  898.      *       | cal_name    |         | category_id   |   <----- | category_name |
  899.      *       ---------------         | is_shown      |          -----------------
  900.      *                               -----------------
  901.      * </pre>
  902.      * In this schema, a Calendar can be assigned to multiple Categories and a Category can have multiple calendars
  903.      * assigned to it. If we want to see which calendars are assigned to a specific category, we have to look at the
  904.      * assignment table. In our code, we would have a Calendar object and a Category object. From the Calendar object
  905.      * we would access the list of Categories by calling something like this:
  906.      * <code>
  907.      *    $categories = $CalA->categories->getAll();
  908.      *    // print list of category names and whether they are going to be shown
  909.      *    foreach ( $categories as $category )
  910.      *        echo $category['object']->category_name . " - " . $category['is_shown'];
  911.      * </code>
  912.      * This would retreive a list of all of the categories along with the properties of the relationship. The objects
  913.      * themselves would be inside the 'object' key of the associative array.
  914.      *
  915.      * In order to set up this relationship, we would make the following call in the object constructor:
  916.      * <code>
  917.      *    $this->_addMultiRelation('categories', 'calendar_id', 'Category', 'category_id', 'CategoryAssign');
  918.      * </code>
  919.      *
  920.      * The other relationship that this class can be used for is a ONE-TO-MANY. For example, in the following schema,
  921.      * we have a ONE-TO-MANY (OTM) relationship between the Events and their Occurrences:
  922.      * <pre>
  923.      *       Events                  Occurrences
  924.      *      ----------------        --------------
  925.      *      | event_id     |<-------| o_event_id |
  926.      *      | event_detail |        | a_date     |
  927.      *      ----------------        --------------
  928.      * </pre>
  929.      * From the Event class we would be looking at a list of multiple Occurrences, and we would set up the relationship
  930.      * like this:
  931.      * <code>
  932.      *    $this->_addMultiRelation('occurrences', 'event_id', 'Occurrence', 'o_event_id');
  933.      * </code>
  934.      * Note that the major difference between the initiation for the ONE-TO-MANY relationship and the MANY-TO-MANY is
  935.      * that there is no associative table in the ONE-TO-MANY one. Beyond that, accessing the entries is exactly the
  936.      * same. For example, to display a list of all of the occurrence dates associated with Event object eventA:
  937.      * <code>
  938.      *    $occurrences = $eventA->occurrences->getAll();
  939.      *    foreach ( $occurrences as $occurrence )
  940.      *        echo $occurrence->a_date;
  941.      * </code>
  942.      * If you wanted to add another occurrence, for example $OccB, then you could call:
  943.      * <code>
  944.      *    $eventA->occurrences->add($OccB);
  945.      * </code>
  946.      *
  947.      * It is important to note that in this example, we are talking about how to implement the Event class. From
  948.      * the perspective of the Occurrence class, this is a simple "single" relationship and will be implemented
  949.      * as described in the _addSingleRelation documentation.
  950.      *
  951.      * @param string $name Name of the relationship. The field name that we will access the relationship by.
  952.      * @param string $local_key Name of the field in this object that we will use relate to the other object.
  953.      * @param string $remote_class Name of the class that the remote (related to) object will be an instance of.
  954.      * @param string $remote_key Name of the field in the remote object that will equal the local key field.
  955.      * @param string $assoc_table Name of the table that contains the relationship information.
  956.      */
  957.     protected function _addMultiRelation($name, $local_key, $remote_class, $remote_key, $assoc_table='') {
  958.         // Create the wrapper
  959.         $this->_relations[$name] = new dbList($this->_db, $assoc_table, $local_key, $this, $remote_key, $remote_class);
  960.     }
  961.     
  962.     /**
  963.      * Provides database wrapped access to the values of the object.
  964.      *
  965.      * This is a magic PHP function and it allows us to control all of the data going in and out of this object.
  966.      * Whenever the user tries to access a property, for example to read "$cal->calendar_id", this function will
  967.      * be called. This allows us to make relationship properties appear as local properties as well as to cache
  968.      * them so we can retrieve them on demand, or ignore them if we don't end up needing them.
  969.      *
  970.      * @param string $name The name of the property we are trying to access
  971.      * @return mixed Value of the property accessed.
  972.  
  973.      */
  974.     public function __get($name) {
  975.         // Handles loading a dbObjProperty one-to-one relation
  976.         if ( $this->_relations[$name] instanceof dbObjProperty ) {
  977.             // Load the object value if not already loaded
  978.             if ( !$this->_relations[$name]->isInit() ) {
  979.                 // Load up the variables needed for referencing
  980.                 $class_name = $this->_relations[$name]->getClass();
  981.                 $local_key  = $this->_relations[$name]->getKey();
  982.                 $local_value = $this->_values[$local_key];
  983.  
  984.                 // Load the object from the database
  985.                 $this->_relations[$name]->set(call_user_func(array($class_name, 'load'), $this->_db, $local_value));
  986.             }
  987.             
  988.             // return reference to the object
  989.             return $this->_relations[$name]->get();
  990.         }
  991.         
  992.         // Handles multi-entry relationships
  993.         else if ( $this->_relations[$name] instanceof dbList ) {
  994.             return $this->_relations[$name];
  995.         }
  996.  
  997.         // Or if it's a basic value, return it directly
  998.         else if ( isset($this->_values[$name]) )
  999.             return $this->_values[$name];
  1000.  
  1001.         // Or if it's not set, return NULL
  1002.         else
  1003.             return NULL;
  1004.     }
  1005.  
  1006.     /**
  1007.      * Database wrapped access method to write values into this object.
  1008.      *
  1009.      * A PHP magic method that allows us to control any attempts to write data into our
  1010.      * object. For example, when we set the title of a Calendar object by calling:
  1011.      * <code>
  1012.      *    $cal->title = "test title";
  1013.      * </code>
  1014.      * this method will be called with the first paramter being set to "title" for the name
  1015.      * of the property and the second paramter to "test title" for the value we are trying
  1016.      * to set it to. This works the same for when we assign objects, etc. This function is
  1017.      * usually not called directly.
  1018.      *
  1019.      * @param string $name Name of the property being set.
  1020.      * @param mixed $value The value being assigned to
  1021.      */    
  1022.     public function __set($name, $value) {
  1023.         // Validate input
  1024.         if ( !isset($this->_types[$name]) && !isset($this->_relations[$name]) )
  1025.             throw new Exception("$name not a valid field of ".get_class($this));
  1026.         if ( $this->_relations[$name] instanceof dbObject && !$value instanceof dbObject )
  1027.             throw new Exception("Formatting error, \$value not a dbObject");
  1028.  
  1029.         // Save the value
  1030.         if ( $this->_relations[$name] instanceof dbObjProperty ) {
  1031.             // Save the object value
  1032.             $this->_relations[$name]->set($value);
  1033.             
  1034.             // Update the local variable the mark the change
  1035.             $key = $this->_relations[$name]->getKey();
  1036.             if ( $key ) {
  1037.                 $this->_values[$key] = $this->_relations[$name]->getKeyValue();
  1038.             
  1039.                 // Update the database
  1040.                 $this->_db->putObject($this);
  1041.             }
  1042.         } else {
  1043.             $this->_values[$name] = $value;
  1044.         
  1045.             // Save to database
  1046.             $this->_db->putObject($this);
  1047.         }
  1048.     }
  1049.  
  1050.     /**
  1051.      * Only to be called by the dbDatabase object. Used to clean up relationships when
  1052.      * we are deleting an object.
  1053.      */
  1054.     public function _delRelations() {
